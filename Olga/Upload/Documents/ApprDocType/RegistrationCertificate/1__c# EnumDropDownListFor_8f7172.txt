 public class PaymentSystem:IEntity<int>
    {
        public PaymentSystem()
        {
            BookAccounts = new List<BookAccount.BookAccount>();
        }
        public int Id { get; set; }

         [StringLength(100)]
         public string Name { get; set; }

         [StringLength(300)]
         public string Comment { get; set; }

         public bool IsDeleted { get; set; } = false;

         public DateTime DateCreated { get; set; }

         public PaymentSystemType PaymentSystemType { get; set; }

         public PaymentSystemStatus PaymentSystemStatus { get; set; }

        public  ICollection<BookAccount.BookAccount> BookAccounts { get; set; }

    }

 public enum PaymentSystemType {
        Email,
        PaymentSystem,
        BookMakerSystem
    }

    public enum PaymentSystemStatus {
        Active,
        Blocked
    }
